# PART A: Document Generation Engine

## Hybrid Approach: Templates + LLM

The engine uses a **two-layer approach**:
1. **Static Templates** — define the mandatory structure, headings, and legal boilerplate
2. **LLM** — fills in fact-specific sections using user-provided data only

This prevents hallucination while maintaining legal accuracy.

## LLM's Role in Document Generation

The LLM is instructed to **ONLY fill** the `{placeholder}` fields using user-provided facts:

```python
def generate_document(case, responses, template):
    # 1. Build facts dict from user responses
    facts = extract_facts(responses)

    # 2. Send to LLM with strict instructions
    prompt = DOCUMENT_PROMPT.format(
        document_type=template["document_type"],
        applicable_laws=case.applicable_laws,
        user_facts=json.dumps(facts),
        template_structure=template["structure"]
    )

    # 3. LLM returns filled document
    result = call_with_retry(llm_client, prompt, DocumentSchema)

    # 4. Post-process: validate no hallucination markers
    validate_document(result, facts)

    return result
```

## Hallucination Prevention

| Technique | Implementation |
|-----------|---------------|
| **Fact anchoring** | LLM is only given user-provided facts; no external lookup |
| **[BLANK] instruction** | LLM must write `[BLANK]` if info is missing, not invent |
| **Law whitelist** | Only pre-approved law citations are injected via prompt, not generated |
| **Output validation** | Pydantic schema checks all required fields are present |
| **Missing info tracking** | `missing_information` field lists all `[BLANK]` items |
| **No external search** | LLM has no web access; only uses prompt context |
| **Disclaimer injection** | Hardcoded disclaimer always appended, never LLM-generated |

## 2. Confidence Scoring

```python
# apps/risk/scoring.py

CONFIDENCE_THRESHOLDS = {
    "high":   (0.80, 1.00),  # Proceed confidently
    "medium": (0.60, 0.79),  # Proceed with caveats
    "low":    (0.40, 0.59),  # Strongly recommend lawyer
    "very_low": (0.0, 0.39)  # Block document generation, flag for review
}

def get_confidence_label(score: float) -> str:
    for label, (low, high) in CONFIDENCE_THRESHOLDS.items():
        if low <= score <= high:
            return label
    return "unknown"

def should_block_generation(score: float) -> bool:
    """Block document generation for very low confidence cases."""
    return score < 0.40
```

## 3. Hallucination Checks

```python
# core/validators/hallucination.py

APPROVED_LAW_PREFIXES = [
    "Indian Penal Code", "IPC", "Bharatiya Nyaya Sanhita", "BNS",
    "Payment of Wages Act", "Industrial Disputes Act",
    "Consumer Protection Act", "IT Act", "Information Technology Act",
    "Transfer of Property Act", "Code of Civil Procedure", "CPC",
    "BNSS", "Code of Criminal Procedure", "CrPC",
    "Contract Act", "Specific Relief Act", "RTI Act",
    "Prevention of Corruption Act", "POCSO Act", "Domestic Violence Act"
]

def validate_law_citations(cited_laws: list) -> tuple[bool, list]:
    """Returns (is_valid, invalid_citations)"""
    invalid = []
    for law in cited_laws:
        if not any(law.startswith(prefix) for prefix in APPROVED_LAW_PREFIXES):
            invalid.append(law)
    return len(invalid) == 0, invalid
```

## 4. Output Validation with Pydantic

```python
# core/validators/schemas.py
from pydantic import BaseModel, field_validator
from typing import List, Optional

class ClassifyResponse(BaseModel):
    primary_category: str
    sub_category: str
    confidence: float
    applicable_laws: List[str]
    jurisdiction: str
    urgency: str
    reasoning: str

    @field_validator('confidence')
    def confidence_range(cls, v):
        if not 0.0 <= v <= 1.0:
            raise ValueError('Confidence must be between 0 and 1')
        return round(v, 3)

    @field_validator('urgency')
    def valid_urgency(cls, v):
        allowed = ['low', 'medium', 'high', 'critical']
        if v not in allowed:
            raise ValueError(f'Urgency must be one of {allowed}')
        return v

    @field_validator('applicable_laws')
    def validate_laws(cls, v):
        is_valid, invalid = validate_law_citations(v)
        if not is_valid:
            raise ValueError(f'Unrecognized law citations: {invalid}')
        return v

class DocumentResponse(BaseModel):
    document_title: str
    document_body: str
    sections_included: List[str]
    missing_information: List[str]
    disclaimer: str
    suggested_send_to: Optional[str]
    filing_deadline_note: Optional[str]

    @field_validator('document_body')
    def check_no_placeholders(cls, v):
        # Warn if LLM left unfilled placeholders (not [BLANK])
        import re
        bad = re.findall(r'\{[a-z_]+\}', v)
        if bad:
            raise ValueError(f'Unfilled template placeholders found: {bad}')
        return v
```

## 5. Input Safety Filter

```python
# core/validators/input_filter.py

BLOCKED_PATTERNS = [
    r"(bomb|explosive|weapon)",           # Violence
    r"(how to kill|murder planning)",     # Criminal intent
    r"(fake document|forge|fabricate)",   # Document fraud
]

def is_safe_input(text: str) -> tuple[bool, str]:
    import re
    for pattern in BLOCKED_PATTERNS:
        if re.search(pattern, text, re.IGNORECASE):
            return False, "Input contains prohibited content."
    return True, ""
```

## 7. Complete Request Lifecycle (DRF View)

```python
# apps/classify/views.py
from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework import status
from core.llm.openai_client import OpenAIClient
from core.llm.retry import call_with_retry
from core.prompts.normalize import NORMALIZE_PROMPT
from core.prompts.classify import CLASSIFY_PROMPT
from core.validators.schemas import ClassifyResponse
from core.validators.input_filter import is_safe_input
from core.utils import inject_disclaimer
from apps.cases.models import Case

class ClassifyView(APIView):
    def post(self, request):
        user_input = request.data.get('user_input', '').strip()

        # 1. Input safety check
        safe, reason = is_safe_input(user_input)
        if not safe:
            return Response({'error': reason}, status=status.HTTP_400_BAD_REQUEST)

        llm = OpenAIClient()

        try:
            # 2. Normalize Hinglish → English
            norm_result = call_with_retry(
                llm,
                NORMALIZE_PROMPT.format(user_input=user_input),
                NormalizeResponse
            )

            # 3. Classify issue
            classify_result = call_with_retry(
                llm,
                CLASSIFY_PROMPT.format(normalized_text=norm_result.normalized_text),
                ClassifyResponse
            )

            # 4. Create Case in DB
            case = Case.objects.create(
                user=request.user,
                raw_input=user_input,
                normalized_text=norm_result.normalized_text,
                primary_category=classify_result.primary_category,
                sub_category=classify_result.sub_category,
                confidence_score=classify_result.confidence,
                applicable_laws=classify_result.applicable_laws,
                jurisdiction=classify_result.jurisdiction,
                urgency=classify_result.urgency,
            )

            response = {
                'case_id': str(case.id),
                **classify_result.model_dump(),
                **norm_result.model_dump()
            }

            return Response(inject_disclaimer(response), status=status.HTTP_201_CREATED)

        except Exception as e:
            return Response({'error': 'Classification failed. Please try again.'}, status=status.HTTP_500_INTERNAL_SERVER_ERROR)
```
